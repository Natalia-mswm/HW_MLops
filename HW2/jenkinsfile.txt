#!/usr/bin/env groovy

pipeline {
    agent any
    options {
        skipDefaultCheckout(true)
    }
    stages {
        stage('CleanWorkspace') {
            steps {
                cleanWs()
                checkout scm
                //echo "Building ${env.MLops_HW2}..."
            }
        }
        stage('Informing') {
            steps {
                echo '--- Environment info ---'
                sh 'python3 --version'
                cleamWs()
                echo 'Working directory was cleaned successfuly'
            }
        }
        stage('Getting repo') {
            steps {
                echo 'clone the repo'
                sh 'git clone https://github.com/Natalia-mswm/HW_MLops.git'
            }
        }
        stage('installing req') {
            steps {
                sh 'python3 -m pip install --upgrade -r ./HW_MLops/HW2/requirements.txt'
            }
        }
        stage('Build') {
            steps {
                echo "Starting data creation"
                sh 'python3 var/lib/jenkins/MLops_HW2/HW2/data_creation.py'
                echo "starting model_preprocessing"
                sh 'python3 var/lib/jenkins/MLops_HW2/HW2/model_preprocessing.py'
                echo 'starting model_preparation'
                sh 'python3 var/lib/jenkins/MLops_HW2/HW2/model_preparation.py'
                echo 'starting model_testing'
                sh 'python3 var/lib/jenkins/MLops_HW2/HW2/model_testing.py'
                echo "Finished!"
            }
        }
    }
}
